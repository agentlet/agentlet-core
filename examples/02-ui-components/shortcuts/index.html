üìú Shortcuts API Examples:

Register a Simple Shortcut

// Register Ctrl+J to show an alert
window.agentlet.utils.shortcuts.register('ctrl+j', () => {
    alert('Hello from Ctrl+J!');
}, {
    description: 'Show hello alert',
    allowInInputs: false
});

Register Multiple Key Combinations

// Register both Ctrl+; and Cmd+; for cross-platform compatibility
window.agentlet.utils.shortcuts.register('ctrl+;,cmd+;', () => {
    window.agentlet.utils.Dialog.commandPrompt({
        title: 'Command Palette',
        placeholder: 'Type a command...'
    }, console.log);
}, {
    description: 'Open command palette',
    allowInInputs: false
});

Advanced Shortcut with Options

// Register F1 for help (works everywhere)
window.agentlet.utils.shortcuts.register('f1', () => {
    window.agentlet.utils.shortcuts.showHelp();
}, {
    description: 'Show shortcuts help',
    allowInInputs: true,  // Works even in input fields
    preventDefault: true,
    stopPropagation: true
});

Manage Shortcuts

// Check if registered
const isRegistered = window.agentlet.utils.shortcuts.isRegistered('ctrl+j');

// Get all shortcuts
const shortcuts = window.agentlet.utils.shortcuts.getShortcuts();

// Unregister a shortcut
window.agentlet.utils.shortcuts.unregister('ctrl+j');

// Enable/disable all shortcuts
window.agentlet.utils.shortcuts.setEnabled(false);
